# Get ALL trajectories
PREFIX kbe:<http://www.my-kbe.com/shapes.owl#>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
  SELECT *
  WHERE
  {
  ?trajectory a kbe:Trajectory.
  ?trajectory kbe:Points ?point.
  ?point kbe:index ?index.
    ?point kbe:x ?x.
    ?point kbe:y ?y.
  }
